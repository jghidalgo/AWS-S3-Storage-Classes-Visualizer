AWS CLI Examples

# Upload file to S3 Standard
aws s3 cp document.pdf s3://my-bucket/

# Upload directly to Standard-IA
aws s3 cp document.pdf s3://my-bucket/ --storage-class STANDARD_IA

# Upload to Glacier
aws s3 cp document.pdf s3://my-bucket/ --storage-class GLACIER

# Upload to Deep Archive
aws s3 cp document.pdf s3://my-bucket/ --storage-class DEEP_ARCHIVE
# Create lifecycle policy JSON
{
  "Rules": [
    {
      "ID": "LifecycleRule",
      "Status": "Enabled",
      "Transitions": [
        {
          "Days": 30,
          "StorageClass": "STANDARD_IA"
        },
        {
          "Days": 90,
          "StorageClass": "GLACIER"
        },
        {
          "Days": 365,
          "StorageClass": "DEEP_ARCHIVE"
        }
      ]
    }
  ]
}

# Apply lifecycle policy
aws s3api put-bucket-lifecycle-configuration \
  --bucket my-bucket \
  --lifecycle-configuration file://lifecycle.json
# Restore from Glacier (expedited: 1-5 min)
aws s3api restore-object \
  --bucket my-bucket \
  --key document.pdf \
  --restore-request Days=1,GlacierJobParameters='{Tier=Expedited}'

# Restore from Glacier (standard: 3-5 hours)
aws s3api restore-object \
  --bucket my-bucket \
  --key document.pdf \
  --restore-request Days=1,GlacierJobParameters='{Tier=Standard}'

# Restore from Deep Archive (standard: 12 hours)
aws s3api restore-object \
  --bucket my-bucket \
  --key document.pdf \
  --restore-request Days=1,GlacierJobParameters='{Tier=Standard}'
